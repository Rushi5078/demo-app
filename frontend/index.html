fetch('http://backend:5000/employees') .then(res => { if (!res.ok) throw new Error('HTTP ' + res.status); return res.json(); }) .then(data => { const ul = document.getElementById('employeeList'); data.forEach(emp => { const li = document.createElement('li'); li.textContent = `${emp.name} (${emp.department}) - $${emp.salary}`; ul.appendChild(li); }); }) .catch(err => { console.error('Fetch failed:', err); });
